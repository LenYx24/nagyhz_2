238inline bool almostEQ(double a, double b) {
239 // eps: ha a relatív, vagy abszolút hiba ettől kisebb, akkor elfogadjuk
240 double eps = 10 * std::numeric_limits<double>::epsilon(); // 10-szer a legkisebb érték
241 if (a == b) return true;
242 if (fabs(a - b) < eps)
243 return true;
244 double aa = fabs(a);
245 double ba = fabs(b);
246 if (aa < ba) {
247 aa = ba;
248 ba = fabs(a);
249 }
250 return (aa - ba) < aa * eps;
251}
